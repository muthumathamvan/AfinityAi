<script>
const KEYWORDS = ["powered by", "elevenlabs", "agents"];
setInterval(() => {
    const w = document.querySelector("elevenlabs-convai");
    if (!w || !w.shadowRoot) return;
    w.shadowRoot.querySelectorAll("*").forEach(el => {
        const t = (el.textContent || "").toLowerCase();
        if (KEYWORDS.some(k => t.includes(k))) el.style.display = "none";
    });
    w.style.visibility = "visible";
}, 300);
</script>
